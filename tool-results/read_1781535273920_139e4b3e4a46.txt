   306тЖТ   306тЖТ    .filter(Boolean) as Player[];

   314тЖТ   314тЖТ
   315тЖТ   315тЖТ  for (const player of startingPlayers) {
   316тЖТ   316тЖТ    const fatigueFactor = 1 - (fatigue.get(player.id) || 0) * 0.003;
   317тЖТ   317тЖТ    const effectiveRating = player.rating * fatigueFactor;
   318тЖТ   318тЖТ
   319тЖТ   319тЖТ    if (ATTACK_POSITIONS.includes(player.position)) {
   320тЖТ   320тЖТ      attackSum += effectiveRating;
   321тЖТ   321тЖТ      attackCount++;
   322тЖТ   322тЖТ    } else if (MID_POSITIONS.includes(player.position)) {
   323тЖТ   323тЖТ      midSum += effectiveRating;
   324тЖТ   324тЖТ      midCount++;
   325тЖТ   325тЖТ    } else {
   326тЖТ   326тЖТ      defSum += effectiveRating;
   327тЖТ   327тЖТ      defCount++;
   328тЖТ   328тЖТ    }
   329тЖТ   329тЖТ  }
   330тЖТ   330тЖТ
   331тЖТ   331тЖТ  const attack = attackCount > 0 ? attackSum / attackCount : team.attackRating;
   332тЖТ   332тЖТ  const midfield = midCount > 0 ? midSum / midCount : team.midfieldRating;
   333тЖТ   333тЖТ  const defense = defCount > 0 ? defSum / defCount : team.defenseRating;
   334тЖТ   334тЖТ  const overall = (attack * 0.35 + midfield * 0.35 + defense * 0.3);
   335тЖТ   335тЖТ
   336тЖТ   336тЖТ  return { attack, midfield, defense, overall };
   337тЖТ   337тЖТ}

   339тЖТ   339тЖТ// тФАтФАтФА Goal Expectancy with Rating-Based Dominance тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   340тЖТ   340тЖТfunction calculateGoalExpectancy(
   341тЖТ   341тЖТ  attackStrength: number,
   342тЖТ   342тЖТ  oppositionDefense: number,
   343тЖТ   343тЖТ  midfieldDiff: number,
   344тЖТ   344тЖТ  tactics: TacticalSettings,
   345тЖТ   345тЖТ  formation: FormationType,
   346тЖТ   346тЖТ  isHome: boolean,
   347тЖТ   347тЖТ  overallRatingDiff: number // New: overall team quality difference
   348тЖТ   348тЖТ): number {
   349тЖТ   349тЖТ  const baseRate = 1.35;
   350тЖТ   350тЖТ
   351тЖТ   351тЖТ  // Enhanced rating-based scaling: bigger gaps = more dominant
   352тЖТ   352тЖТ  const attackRatio = attackStrength / 80;
   353тЖТ   353тЖТ  const defenseRatio = oppositionDefense / 80;
   354тЖТ   354тЖТ
   355тЖТ   355тЖТ  // Rating dominance multiplier: if a team is significantly better, they score more
   356тЖТ   356тЖТ  // Exponential effect makes gaps more decisive
   357тЖТ   357тЖТ  const ratingDominance = Math.pow(1.02, overallRatingDiff);
   358тЖТ   358тЖТ
   359тЖТ   359тЖТ  const formationMod = getFormationAttackBonus(formation);
   360тЖТ   360тЖТ  const mentalityMod = getMentalityModifier(tactics.mentality);
   361тЖТ   361тЖТ  const pressingMod = (tactics.pressingIntensity - 50) / 500;
   362тЖТ   362тЖТ  const tempoMod = (tactics.tempo - 50) / 800;
   363тЖТ   363тЖТ  const homeMod = isHome ? 0.12 : -0.05;
   364тЖТ   364тЖТ  const midMod = midfieldDiff / 500;
   365тЖТ   365тЖТ
   366тЖТ   366тЖТ  const expectancy =
   367тЖТ   367тЖТ    baseRate *
   368тЖТ   368тЖТ    attackRatio *
   369тЖТ   369тЖТ    (2 - defenseRatio) *
   370тЖТ   370тЖТ    ratingDominance *
   371тЖТ   371тЖТ    (1 + formationMod + mentalityMod.attackMod + pressingMod + tempoMod + homeMod + midMod);
   372тЖТ   372тЖТ
   373тЖТ   373тЖТ  return Math.max(0.15, Math.min(4.0, expectancy));
   374тЖТ   374тЖТ}

   376тЖТ   376тЖТfunction poissonRandom(lambda: number, rng: SeededRandom): number {
   377тЖТ   377тЖТ  let L = Math.exp(-lambda);
   378тЖТ   378тЖТ  let k = 0;
   379тЖТ   379тЖТ  let p = 1;
   380тЖТ   380тЖТ  do {
   381тЖТ   381тЖТ    k++;
   382тЖТ   382тЖТ    p *= rng.next();
   383тЖТ   383тЖТ  } while (p > L);
   384тЖТ   384тЖТ  return k - 1;
   385тЖТ   385тЖТ}

   413тЖТ   413тЖТ          description: `Substitution: ${playerOut?.name || sub.playerOutId} тЮЬ ${playerIn?.name || sub.playerInId}`,

   428тЖТ   428тЖТfunction chooseAttackSide(
   429тЖТ   429тЖТ  isTeamA: boolean,
   430тЖТ   430тЖТ  matchupAnalysis: TeamMatchupAnalysis,
   431тЖТ   431тЖТ  rng: SeededRandom
   432тЖТ   432тЖТ): 'left' | 'right' | 'center' {
   433тЖТ   433тЖТ  // Base weights
   434тЖТ   434тЖТ  let leftWeight = 30;
   435тЖТ   435тЖТ  let rightWeight = 30;
   436тЖТ   436тЖТ  let centerWeight = 40;
   437тЖТ   437тЖТ
   438тЖТ   438тЖТ  if (isTeamA) {
   439тЖТ   439тЖТ    // Team A attacks: positive advantage = A's attackers beat B's defenders
   440тЖТ   440тЖТ    leftWeight += matchupAnalysis.leftAdvantage * 150;
   441тЖТ   441тЖТ    rightWeight += matchupAnalysis.rightAdvantage * 150;
   442тЖТ   442тЖТ    centerWeight += matchupAnalysis.centerAdvantage * 150;
   443тЖТ   443тЖТ  } else {
   444тЖТ   444тЖТ    // Team B attacks: negative advantage (from A's perspective) means B has the edge
   445тЖТ   445тЖТ    leftWeight -= matchupAnalysis.leftAdvantage * 150;
   446тЖТ   446тЖТ    rightWeight -= matchupAnalysis.rightAdvantage * 150;
   447тЖТ   447тЖТ    centerWeight -= matchupAnalysis.centerAdvantage * 150;
   448тЖТ   448тЖТ  }
   449тЖТ   449тЖТ
   450тЖТ   450тЖТ  // Ensure minimum weights
   451тЖТ   451тЖТ  leftWeight = Math.max(10, leftWeight);
   452тЖТ   452тЖТ  rightWeight = Math.max(10, rightWeight);
   453тЖТ   453тЖТ  centerWeight = Math.max(10, centerWeight);
   454тЖТ   454тЖТ
   455тЖТ   455тЖТ  const total = leftWeight + rightWeight + centerWeight;
   456тЖТ   456тЖТ  const roll = rng.next() * total;
   457тЖТ   457тЖТ
   458тЖТ   458тЖТ  if (roll < leftWeight) return 'left';
   459тЖТ   459тЖТ  if (roll < leftWeight + rightWeight) return 'right';
   460тЖТ   460тЖТ  return 'center';
   461тЖТ   461тЖТ}

   479тЖТ   479тЖТfunction selectPlayerForAction(
   480тЖТ   480тЖТ  lineup: Lineup,
   481тЖТ   481тЖТ  team: Team,
   482тЖТ   482тЖТ  side: 'left' | 'right' | 'center',
   483тЖТ   483тЖТ  actionType: 'attack' | 'defend' | 'midfield',
   484тЖТ   484тЖТ  rng: SeededRandom,
   485тЖТ   485тЖТ  fatigue: Map<string, number>
   486тЖТ   486тЖТ): Player | null {
   487тЖТ   487тЖТ  const players = lineup.starting11
   488тЖТ   488тЖТ    .map(id => team.players.find(p => p.id === id))
   489тЖТ   489тЖТ    .filter((p): p is Player => !!p);
   490тЖТ   490тЖТ
   491тЖТ   491тЖТ  // Score each player based on relevance to the action
   492тЖТ   492тЖТ  const scored = players.map(player => {
   493тЖТ   493тЖТ    let score = 0;
   494тЖТ   494тЖТ    const fatigueFactor = 1 - (fatigue.get(player.id) || 0) * 0.003;
   495тЖТ   495тЖТ    const effectiveRating = player.rating * fatigueFactor;
   496тЖТ   496тЖТ
   497тЖТ   497тЖТ    // Position relevance
   498тЖТ   498тЖТ    if (actionType === 'attack') {
   499тЖТ   499тЖТ      if (side === 'left' && ['LW', 'LM', 'ST', 'CF'].includes(player.position)) score += 30;
   500тЖТ   500тЖТ      if (side === 'right' && ['RW', 'RM', 'ST', 'CF'].includes(player.position)) score += 30;
   501тЖТ   501тЖТ      if (side === 'center' && ['ST', 'CF', 'CAM'].includes(player.position)) score += 30;
   502тЖТ   502тЖТ      if (ATTACK_POSITIONS.includes(player.position)) score += 20;
   503тЖТ   503тЖТ      if (MID_POSITIONS.includes(player.position)) score += 10;
   504тЖТ   504тЖТ    } else if (actionType === 'defend') {
   505тЖТ   505тЖТ      if (DEF_POSITIONS.includes(player.position)) score += 20;
   506тЖТ   506тЖТ      if (player.position === 'GK') score += 15;
   507тЖТ   507тЖТ      if (['CDM'].includes(player.position)) score += 15;
   508тЖТ   508тЖТ    } else {
   509тЖТ   509тЖТ      if (MID_POSITIONS.includes(player.position)) score += 25;
   510тЖТ   510тЖТ      if (['CDM'].includes(player.position)) score += 15;
   511тЖТ   511тЖТ    }
   512тЖТ   512тЖТ
   513тЖТ   513тЖТ    // Rating bonus
   514тЖТ   514тЖТ    score += (effectiveRating - 60) * 0.5;
   515тЖТ   515тЖТ
   516тЖТ   516тЖТ    return { player, score: Math.max(0, score) };
   517тЖТ   517тЖТ  });
   518тЖТ   518тЖТ
   519тЖТ   519тЖТ  // Weighted random selection
   520тЖТ   520тЖТ  const totalScore = scored.reduce((s, p) => s + p.score, 0);
   521тЖТ   521тЖТ  if (totalScore === 0) return players[0] || null;
   522тЖТ   522тЖТ
   523тЖТ   523тЖТ  let roll = rng.next() * totalScore;
   524тЖТ   524тЖТ  for (const { player, score } of scored) {
   525тЖТ   525тЖТ    roll -= score;
   526тЖТ   526тЖТ    if (roll <= 0) return player;
   527тЖТ   527тЖТ  }
   528тЖТ   528тЖТ
   529тЖТ   529тЖТ  return players[0] || null;
   530тЖТ   530тЖТ}

   532тЖТ   532тЖТ// тФАтФАтФА Main Simulation тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   533тЖТ   533тЖТexport function simulateMatch(
   534тЖТ   534тЖТ  teamA: Team,
   535тЖТ   535тЖТ  teamB: Team,
   536тЖТ   536тЖТ  lineupA: Lineup,
   537тЖТ   537тЖТ  lineupB: Lineup,
   538тЖТ   538тЖТ  seed: number = Date.now()
   539тЖТ   539тЖТ): MatchState {
   540тЖТ   540тЖТ  const rng = new SeededRandom(seed);
   541тЖТ   541тЖТ  const fatigue = new Map<string, number>();
   542тЖТ   542тЖТ  const events: MatchEvent[] = [];
   543тЖТ   543тЖТ
   544тЖТ   544тЖТ  let currentLineupA = { ...lineupA, starting11: [...lineupA.starting11], substitutions: [...lineupA.substitutions] };
   545тЖТ   545тЖТ  let currentLineupB = { ...lineupB, starting11: [...lineupB.starting11], substitutions: [...lineupB.substitutions] };
   546тЖТ   546тЖТ
   547тЖТ   547тЖТ  // Initialize fatigue
   548тЖТ   548тЖТ  [...currentLineupA.starting11, ...currentLineupB.starting11].forEach(id => fatigue.set(id, 0));
   549тЖТ   549тЖТ
   550тЖТ   550тЖТ  // Calculate team strengths
   551тЖТ   551тЖТ  const strengthA = calculateTeamStrength(teamA, currentLineupA, fatigue);
   552тЖТ   552тЖТ  const strengthB = calculateTeamStrength(teamB, currentLineupB, fatigue);
   553тЖТ   553тЖТ
   554тЖТ   554тЖТ  // Overall rating difference for dominance calculations
   555тЖТ   555тЖТ  const overallRatingDiff = strengthA.overall - strengthB.overall;
   556тЖТ   556тЖТ
   557тЖТ   557тЖТ  // Calculate matchups
   558тЖТ   558тЖТ  const matchupAnalysis = calculateMatchups(teamA, teamB, currentLineupA, currentLineupB, fatigue);
   559тЖТ   559тЖТ
   560тЖТ   560тЖТ  // Goal expectancy with enhanced rating dominance
   561тЖТ   561тЖТ  const expA = calculateGoalExpectancy(
   562тЖТ   562тЖТ    strengthA.attack, strengthB.defense, strengthA.midfield - strengthB.midfield,
   563тЖТ   563тЖТ    lineupA.tactics, lineupA.formation, true, overallRatingDiff
   564тЖТ   564тЖТ  );
   565тЖТ   565тЖТ  const expB = calculateGoalExpectancy(
   566тЖТ   566тЖТ    strengthB.attack, strengthA.defense, strengthB.midfield - strengthA.midfield,
   567тЖТ   567тЖТ    lineupB.tactics, lineupB.formation, false, -overallRatingDiff
   568тЖТ   568тЖТ  );
   569тЖТ   569тЖТ
   570тЖТ   570тЖТ  const goalsA = poissonRandom(expA, rng);
   571тЖТ   571тЖТ  const goalsB = poissonRandom(expB, rng);
   572тЖТ   572тЖТ
   573тЖТ   573тЖТ  // Generate goal minutes
   574тЖТ   574тЖТ  const goalMinutesA: number[] = [];
   575тЖТ   575тЖТ  const goalMinutesB: number[] = [];
   576тЖТ   576тЖТ  for (let i = 0; i < goalsA; i++) {
   577тЖТ   577тЖТ    goalMinutesA.push(rng.int(1, 90));
   578тЖТ   578тЖТ  }
   579тЖТ   579тЖТ  for (let i = 0; i < goalsB; i++) {
   580тЖТ   580тЖТ    goalMinutesB.push(rng.int(1, 90));
   581тЖТ   581тЖТ  }
   582тЖТ   582тЖТ  goalMinutesA.sort((a, b) => a - b);
   583тЖТ   583тЖТ  goalMinutesB.sort((a, b) => a - b);
   584тЖТ   584тЖТ
   585тЖТ   585тЖТ  // Possession strength - stronger team dominates more
   586тЖТ   586тЖТ  const possessionStrength = 0.5 + (strengthA.midfield - strengthB.midfield) / 200;
   587тЖТ   587тЖТ
   588тЖТ   588тЖТ  // Track stats
   589тЖТ   589тЖТ  let score: [number, number] = [0, 0];
   590тЖТ   590тЖТ  let shots: [number, number] = [0, 0];
   591тЖТ   591тЖТ  let shotsOnTarget: [number, number] = [0, 0];
   592тЖТ   592тЖТ  let corners: [number, number] = [0, 0];
   593тЖТ   593тЖТ  let fouls: [number, number] = [0, 0];
   594тЖТ   594тЖТ  let yellowCards: [number, number] = [0, 0];
   595тЖТ   595тЖТ  let redCards: [number, number] = [0, 0];
   596тЖТ   596тЖТ  let passes: [number, number] = [0, 0];
   597тЖТ   597тЖТ  let possessionCount: [number, number] = [0, 0];
   598тЖТ   598тЖТ
   599тЖТ   599тЖТ  // Ball tracking
   600тЖТ   600тЖТ  let ballX = 50, ballY = 50;
   601тЖТ   601тЖТ  let ballHolderTeamId = rng.chance(possessionStrength) ? teamA.id : teamB.id;
   602тЖТ   602тЖТ
   603тЖТ   603тЖТ  // Kick off
   604тЖТ   604тЖТ  events.push({
   605тЖТ   605тЖТ    minute: 0,
   606тЖТ   606тЖТ    type: 'kick_off',
   607тЖТ   607тЖТ    teamId: teamA.id,
   608тЖТ   608тЖТ    description: 'Kick off! The match begins.',
   609тЖТ   609тЖТ    x: 50,
   610тЖТ   610тЖТ    y: 50,
   611тЖТ   611тЖТ  });
   612тЖТ   612тЖТ
   613тЖТ   613тЖТ  // тФАтФАтФА Minute-by-Minute Simulation тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   614тЖТ   614тЖТ  for (let minute = 1; minute <= 90; minute++) {
   615тЖТ   615тЖТ    // Fatigue increases - more for high-tempo teams
   616тЖТ   616тЖТ    const tempoFactorA = 1 + (lineupA.tactics.tempo - 50) / 500;
   617тЖТ   617тЖТ    const tempoFactorB = 1 + (lineupB.tactics.tempo - 50) / 500;

  1016тЖТ  1016тЖТfunction generatePlayerPositions(
  1017тЖТ  1017тЖТ  team: Team,
  1018тЖТ  1018тЖТ  lineup: Lineup,
  1019тЖТ  1019тЖТ  ballX: number,
  1020тЖТ  1020тЖТ  ballY: number,
  1021тЖТ  1021тЖТ  isTeamA: boolean,

  1024тЖТ  1024тЖТ  const formation = FORMATIONS[lineup.formation];
  1025тЖТ  1025тЖТ  if (!formation) return [];
  1026тЖТ  1026тЖТ
  1027тЖТ  1027тЖТ  const positions: PlayerPosition[] = [];
  1028тЖТ  1028тЖТ  const defLineShift = (tactics.defensiveLine - 50) / 200;
  1029тЖТ  1029тЖТ  const widthShift = (tactics.width - 50) / 400;
  1030тЖТ  1030тЖТ  const pressShift = (tactics.pressingIntensity - 50) / 300;
  1031тЖТ  1031тЖТ
  1032тЖТ  1032тЖТ  // Team-wide shift based on ball position
  1033тЖТ  1033тЖТ  // When ball is in attacking half, the whole team shifts up
  1034тЖТ  1034тЖТ  const teamShiftX = isTeamA
  1035тЖТ  1035тЖТ    ? (ballX > 50 ? (ballX - 50) * 0.08 : 0)
  1036тЖТ  1036тЖТ    : (ballX < 50 ? (50 - ballX) * 0.08 : 0);
  1037тЖТ  1037тЖТ
  1038тЖТ  1038тЖТ  // Team-wide lateral shift when ball is on one side
  1039тЖТ  1039тЖТ  const teamShiftY = (ballY - 50) * 0.05;
  1040тЖТ  1040тЖТ
  1041тЖТ  1041тЖТ  for (let i = 0; i < Math.min(11, lineup.starting11.length); i++) {
  1042тЖТ  1042тЖТ    const slot = formation[i];
  1043тЖТ  1043тЖТ    const playerId = lineup.starting11[i];
  1044тЖТ  1044тЖТ    if (!slot || !playerId) continue;
  1045тЖТ  1045тЖТ
  1046тЖТ  1046тЖТ    const player = team.players.find(p => p.id === playerId);
  1047тЖТ  1047тЖТ
  1048тЖТ  1048тЖТ    let baseX = slot.x + defLineShift * 15;
  1049тЖТ  1049тЖТ    let baseY = slot.y + widthShift * 10;
  1050тЖТ  1050тЖТ
  1051тЖТ  1051тЖТ    // Position-specific ball attraction
  1052тЖТ  1052тЖТ    let ballAttrX: number, ballAttrY: number;
  1053тЖТ  1053тЖТ
  1054тЖТ  1054тЖТ    if (player && ATTACK_POSITIONS.includes(player.position)) {
  1055тЖТ  1055тЖТ      // Attackers: strongly attracted to ball, push forward
  1056тЖТ  1056тЖТ      ballAttrX = (ballX - baseX) * 0.2 + pressShift * 10;
  1057тЖТ  1057тЖТ      ballAttrY = (ballY - baseY) * 0.15;
  1058тЖТ  1058тЖТ      // Extra forward push when team has possession on attack side
  1059тЖТ  1059тЖТ      if (isTeamA && ballX > 50) {
  1060тЖТ  1060тЖТ        ballAttrX += (ballX - 50) * 0.1;
  1061тЖТ  1061тЖТ      } else if (!isTeamA && ballX < 50) {
  1062тЖТ  1062тЖТ        ballAttrX += (50 - ballX) * 0.1;
  1063тЖТ  1063тЖТ      }
  1064тЖТ  1064тЖТ    } else if (player && MID_POSITIONS.includes(player.position)) {
  1065тЖТ  1065тЖТ      // Midfielders: moderate attraction, balanced
  1066тЖТ  1066тЖТ      ballAttrX = (ballX - baseX) * 0.15 + pressShift * 8;
  1067тЖТ  1067тЖТ      ballAttrY = (ballY - baseY) * 0.12;
  1068тЖТ  1068тЖТ    } else if (player && player.position === 'GK') {
  1069тЖТ  1069тЖТ      // Goalkeeper: stays in goal area, slight lateral movement
  1070тЖТ  1070тЖТ      ballAttrX = (ballX - baseX) * 0.02;
  1071тЖТ  1071тЖТ      ballAttrY = (ballY - baseY) * 0.05;
  1072тЖТ  1072тЖТ    } else {
  1073тЖТ  1073тЖТ      // Defenders: moderate forward shift but stay behind ball
  1074тЖТ  1074тЖТ      ballAttrX = (ballX - baseX) * 0.1 + pressShift * 6;
  1075тЖТ  1075тЖТ      ballAttrY = (ballY - baseY) * 0.1;
  1076тЖТ  1076тЖТ      // Defenders shouldn't push too far forward
  1077тЖТ  1077тЖТ      if (isTeamA && baseX + ballAttrX > 60) {
  1078тЖТ  1078тЖТ        ballAttrX = Math.min(ballAttrX, 60 - baseX);
  1079тЖТ  1079тЖТ      } else if (!isTeamA && baseX + ballAttrX < 40) {
  1080тЖТ  1080тЖТ        ballAttrX = Math.max(ballAttrX, 40 - baseX);
  1081тЖТ  1081тЖТ      }
  1082тЖТ  1082тЖТ    }
  1083тЖТ  1083тЖТ
  1084тЖТ  1084тЖТ    baseX += ballAttrX + teamShiftX;
  1085тЖТ  1085тЖТ    baseY += ballAttrY + teamShiftY;
  1086тЖТ  1086тЖТ
  1087тЖТ  1087тЖТ    // Clamp positions
  1088тЖТ  1088тЖТ    baseX = Math.max(2, Math.min(98, baseX));
  1089тЖТ  1089тЖТ    baseY = Math.max(3, Math.min(97, baseY));
  1090тЖТ  1090тЖТ
  1091тЖТ  1091тЖТ    if (!isTeamA) {
  1092тЖТ  1092тЖТ      const mirrored = mirrorPosition(baseX, baseY);
  1093тЖТ  1093тЖТ      baseX = mirrored.x;
  1094тЖТ  1094тЖТ      baseY = mirrored.y;
  1095тЖТ  1095тЖТ    }
  1096тЖТ  1096тЖТ
  1097тЖТ  1097тЖТ    // Check if this player has the ball
  1098тЖТ  1098тЖТ    const hasBall = false; // Will be set during animation
  1099тЖТ  1099тЖТ
  1100тЖТ  1100тЖТ    positions.push({
  1101тЖТ  1101тЖТ      playerId,
  1102тЖТ  1102тЖТ      baseX,
  1103тЖТ  1103тЖТ      baseY,
  1104тЖТ  1104тЖТ      currentX: baseX,
  1105тЖТ  1105тЖТ      currentY: baseY,
  1106тЖТ  1106тЖТ      hasBall,
  1107тЖТ  1107тЖТ    });
  1108тЖТ  1108тЖТ  }
  1109тЖТ  1109тЖТ
  1110тЖТ  1110тЖТ  return positions;
  1111тЖТ  1111тЖТ}

  1386тЖТ  1386тЖТ// Generate player positions with per-frame jitter for more realistic movement
  1387тЖТ  1387тЖТfunction generatePlayerPositionsWithJitter(
  1388тЖТ  1388тЖТ  team: Team,
  1389тЖТ  1389тЖТ  lineup: Lineup,
  1390тЖТ  1390тЖТ  ballX: number,
  1391тЖТ  1391тЖТ  ballY: number,
  1392тЖТ  1392тЖТ  isTeamA: boolean,

  1394тЖТ  1394тЖТ  rng: SeededRandom
  1395тЖТ  1395тЖТ): PlayerPosition[] {
  1396тЖТ  1396тЖТ  const formation = FORMATIONS[lineup.formation];

  1398тЖТ  1398тЖТ
  1399тЖТ  1399тЖТ  const positions: PlayerPosition[] = [];
  1400тЖТ  1400тЖТ  const defLineShift = (tactics.defensiveLine - 50) / 200;
  1401тЖТ  1401тЖТ  const widthShift = (tactics.width - 50) / 400;
  1402тЖТ  1402тЖТ  const pressShift = (tactics.pressingIntensity - 50) / 300;
  1403тЖТ  1403тЖТ
  1404тЖТ  1404тЖТ  // Team-wide shift based on ball position
  1405тЖТ  1405тЖТ  const teamShiftX = isTeamA
  1406тЖТ  1406тЖТ    ? (ballX > 50 ? (ballX - 50) * 0.08 : 0)
  1407тЖТ  1407тЖТ    : (ballX < 50 ? (50 - ballX) * 0.08 : 0);
  1408тЖТ  1408тЖТ  const teamShiftY = (ballY - 50) * 0.05;
  1409тЖТ  1409тЖТ
  1410тЖТ  1410тЖТ  for (let i = 0; i < Math.min(11, lineup.starting11.length); i++) {
  1411тЖТ  1411тЖТ    const slot = formation[i];
  1412тЖТ  1412тЖТ    const playerId = lineup.starting11[i];
  1413тЖТ  1413тЖТ    if (!slot || !playerId) continue;
  1414тЖТ  1414тЖТ
  1415тЖТ  1415тЖТ    const player = team.players.find(p => p.id === playerId);
  1416тЖТ  1416тЖТ
  1417тЖТ  1417тЖТ    let baseX = slot.x + defLineShift * 15;
  1418тЖТ  1418тЖТ    let baseY = slot.y + widthShift * 10;
  1419тЖТ  1419тЖТ

  1424тЖТ  1424тЖТ      // Attackers: strongly attracted to ball, push forward

  1426тЖТ  1426тЖТ      ballAttrY = (ballY - baseY) * 0.15;

  1428тЖТ  1428тЖТ        ballAttrX += (ballX - 50) * 0.12;
  1429тЖТ  1429тЖТ      } else if (!isTeamA && ballX < 50) {
  1430тЖТ  1430тЖТ        ballAttrX += (50 - ballX) * 0.12;

  1432тЖТ  1432тЖТ    } else if (player && MID_POSITIONS.includes(player.position)) {
  1433тЖТ  1433тЖТ      // Midfielders: moderate attraction
  1434тЖТ  1434тЖТ      ballAttrX = (ballX - baseX) * 0.16 + pressShift * 8;
  1435тЖТ  1435тЖТ      ballAttrY = (ballY - baseY) * 0.12;

  1458тЖТ  1458тЖТ    baseY += jitterY;
  1459тЖТ  1459тЖТ
  1460тЖТ  1460тЖТ    // Clamp

  1474тЖТ  1474тЖТ      playerId,
  1475тЖТ  1475тЖТ      baseX,
  1476тЖТ  1476тЖТ      baseY,
  1477тЖТ  1477тЖТ      currentX: baseX,
  1478тЖТ  1478тЖТ      currentY: baseY,
  1479тЖТ  1479тЖТ      hasBall,

  1481тЖТ  1481тЖТ  }
  1482тЖТ  1482тЖТ
